2. Writing to Files - StreamWriter

Append Mode (add to end of file)

// true parameter = append mode (doesn't overwrite existing content)
using (StreamWriter sw = new StreamWriter(filePath, true))
{
    sw.WriteLine(line);  // Writes line and adds newline at the end
}

Overwrite Mode (replace entire file)

// false parameter = overwrite mode
using (StreamWriter sw = new StreamWriter(filePath, false))
{
    sw.WriteLine(line);
}

Complete Example: Save New Product

static void SaveProduct()
{
    Console.WriteLine("\n---- Enter New Product ----");

    Console.Write("Product name: ");
    string name = Console.ReadLine();

    Console.Write("Price: ");
    string price = Console.ReadLine();

    Console.Write("Stock: ");
    string stock = Console.ReadLine();

    Console.Write("Supplier: ");
    string supplier = Console.ReadLine();

    // Create line with all data, separated by semicolon
    string line = $"{name};{price};{stock};{supplier}";

    try
    {
        // StreamWriter with true = append mode
        using (StreamWriter sw = new StreamWriter("products.txt", true))
        {
            sw.WriteLine(line);
        }
        Console.WriteLine("\nProduct successfully saved!");
    }
    catch (Exception ex)
    {
        Console.WriteLine($"\nError occurred: {ex.Message}");
    }
}
Key Concepts:

3. Updating Products in TXT File

This example shows how to edit an existing product inside a TXT file. Logic:

  • Read all products into a list
  • Find product by name
  • Modify its fields
  • Rewrite entire file with updated data
static void EditProduct()
{
    if (!File.Exists("products.txt"))
    {
        Console.WriteLine("The products file does not exist!");
        return;
    }

    Console.WriteLine("\n ---- EDIT PRODUCT ----");
    Console.WriteLine("Enter the product name you want to edit: ");
    string searchedName = Console.ReadLine();

    List updatedLines = new List();
    bool found = false;

    try
    {
        using (StreamReader sr = new StreamReader("products.txt"))
        {
            string line;

            while ((line = sr.ReadLine()) != null)
            {
                string[] data = line.Split(';');
                if (data.Length == 4 && data[0].ToLower() == searchedName.ToLower())
                {
                    Console.WriteLine($"Product found: {data[0]}");
                    Console.WriteLine($"Current price: {data[1]} EUR");
                    Console.WriteLine($"Current stock: {data[2]}");
                    Console.WriteLine($"Current supplier: {data[3]}");

                    string newPrice = data[1];
                    Console.Write("\nEnter new price (leave empty to keep current): ");
                    string newPriceStr = Console.ReadLine();
                    if (!string.IsNullOrWhiteSpace(newPriceStr)) newPrice = newPriceStr;

                    string newStock = data[2];
                    Console.Write("Enter new stock (leave empty to keep current): ");
                    string newStockStr = Console.ReadLine();
                    if (!string.IsNullOrWhiteSpace(newStockStr)) newStock = newStockStr;

                    string newSupplier = data[3];
                    Console.Write("Enter new supplier (leave empty to keep current): ");
                    string newSupplierStr = Console.ReadLine();
                    if (!string.IsNullOrWhiteSpace(newSupplierStr)) newSupplier = newSupplierStr;

                    string newLine = $"{data[0]};{newPrice};{newStock};{newSupplier}";
                    updatedLines.Add(newLine);

                    found = true;
                }
                else
                {
                    updatedLines.Add(line);
                }
            }
        }

        if (found)
        {
            using (StreamWriter sw = new StreamWriter("products.txt", false))
            {
                foreach (string l in updatedLines)
                {
                    sw.WriteLine(l);
                }
            }

            Console.WriteLine("Product successfully updated!");
        }
        else
        {
            Console.WriteLine("A product with that name was not found!");
        }
    }
    catch (Exception ex)
    {
        Console.WriteLine($"An error has occurred: {ex.Message}");
    }
}

5. Searching & Filtering Data

Complete Example: Search Products by Supplier and Stock

static void SearchProducts()
{
    if (!File.Exists("products.txt"))
    {
        Console.WriteLine("\nFile doesn't exist!");
        return;
    }

    Console.WriteLine("\n---- SEARCH PRODUCTS ----");
    
    Console.Write("Enter supplier name: ");
    string searchSupplier = Console.ReadLine();

    Console.Write("Enter maximum stock: ");
    string maxStockInput = Console.ReadLine();
    int maxStock = int.Parse(maxStockInput);

    // List to store found products
    List foundProducts = new List();

    try
    {
        using (StreamReader sr = new StreamReader("products.txt"))
        {
            string line;

            while ((line = sr.ReadLine()) != null)
            {
                // Split line into parts by semicolon
                string[] data = line.Split(';');

                if (data.Length == 4)
                {
                    string name = data[0];
                    string price = data[1];
                    int stock = int.Parse(data[2]);
                    string supplier = data[3];

                    // Check if supplier matches AND stock is less than max
                    if ((supplier.ToLower() == searchSupplier.ToLower()) 
                        && (stock < maxStock))
                    {
                        foundProducts.Add(line);
                    }
                }
            }

            // Display results
            if (foundProducts.Count > 0)
            {
                Console.WriteLine($"\nFound {foundProducts.Count} products:");
                Console.WriteLine("-------------------------------");

                foreach (string product in foundProducts)
                {
                    string[] data = product.Split(";");
                    Console.WriteLine($"Product: {data[0]}");
                    Console.WriteLine($"Price: {data[1]} EUR");
                    Console.WriteLine($"Stock: {data[2]} units");
                    Console.WriteLine($"Supplier: {data[3]}");
                    Console.WriteLine("-------------------------------");
                }

                // Save results to separate file
                string resultsFile = "search_results.txt";
                using (StreamWriter sw = new StreamWriter(resultsFile, false))
                {
                    sw.WriteLine($"SEARCH RESULTS - Supplier: {searchSupplier}, Stock < {maxStock}");
                    sw.WriteLine("================================");

                    foreach (string product in foundProducts)
                    {
                        sw.WriteLine(product);
                    }
                }
                Console.WriteLine($"\nResults saved to {resultsFile}");
            }
            else
            {
                Console.WriteLine("\nNo products found matching criteria.");
            }
        }
    }
    catch (Exception ex)
    {
        Console.WriteLine($"Error occurred while reading: {ex.Message}");
    }
}
Key Techniques:
â€¢ Use List<string> to collect matching items
â€¢ ToLower() for case-insensitive comparison
â€¢ int.Parse() to convert string to number
â€¢ Save filtered results to new file

6. Data Processing & Modification

Complete Example: Discount Prices

static void DiscountPrices()
{
    if (!File.Exists("products.txt"))
    {
        Console.WriteLine("Products file doesn't exist!");
        return;
    }

    Console.WriteLine("\n---- PRICE DISCOUNT ----");
    
    Console.Write("Enter discount percentage (e.g. 10 for 10%): ");
    string percentInput = Console.ReadLine();
    double percent = double.Parse(percentInput);

    if (percent <= 0 || percent > 100)
    {
        Console.WriteLine("\nPercentage must be between 0 and 100");
        return;
    }

    List discountedProducts = new List();
    int productCount = 0;

    try
    {
        // Read all products
        using (StreamReader sr = new StreamReader("products.txt"))
        {
            string line;

            while ((line = sr.ReadLine()) != null)
            {
                string[] data = line.Split(";");
                
                if (data.Length == 4)
                {
                    string name = data[0];
                    double oldPrice = double.Parse(data[1]);
                    string stock = data[2];
                    string supplier = data[3];

                    // Calculate new price
                    double newPrice = oldPrice * (1 - percent / 100);
                    newPrice = Math.Round(newPrice, 2);

                    // Create new line with updated price
                    string newLine = $"{name};{newPrice.ToString("F2")};{stock};{supplier}";

                    discountedProducts.Add(newLine);
                    productCount++;

                    // Display changes
                    Console.WriteLine($"\n{name}");
                    Console.WriteLine($"    Old price: {oldPrice.ToString("F2")} EUR");
                    Console.WriteLine($"    New price: {newPrice.ToString("F2")} EUR, discounted by {percent}%");
                }
            }
        }

        // Save discounted products to new file
        string discountFile = "products_sale.txt";

        using (StreamWriter sw = new StreamWriter(discountFile, false))
        {
            sw.WriteLine($"SALE - Discount {percent}%");
            sw.WriteLine($"Date: {DateTime.Now.ToString("dd.MM.yyyy HH:mm")}");
            sw.WriteLine($"Number of products: {productCount}");
            sw.WriteLine();

            foreach (string product in discountedProducts)
            {
                sw.WriteLine(product);
            }
        }

        Console.WriteLine($"\n\nSuccessfully discounted {productCount} products");
        Console.WriteLine($"Sale products saved to {discountFile}");
    }
    catch (FormatException)
    {
        Console.WriteLine("\nError: Price in file is not a valid number!");
    }
    catch (Exception ex)
    {
        Console.WriteLine($"\nError occurred: {ex.Message}");
    }
}
Important Points:
â€¢ double.Parse() for decimal numbers
â€¢ Math.Round(value, 2) rounds to 2 decimal places
â€¢ ToString("F2") formats number with 2 decimals
â€¢ FormatException catches parsing errors
â€¢ DateTime.Now gets current date and time

7. Common Patterns & Best Practices

String Interpolation for Building Lines

// Modern way (recommended)
string line = $"{name};{price};{stock};{supplier}";

// Old way
string line = name + ";" + price + ";" + stock + ";" + supplier;

Safe Number Parsing

// Using TryParse (safer - doesn't throw exception)
if (int.TryParse(input, out int result))
{
    // Successfully parsed
    Console.WriteLine($"Number: {result}");
}
else
{
    Console.WriteLine("Invalid number!");
}

// Using Parse (throws exception on error)
try
{
    int result = int.Parse(input);
}
catch (FormatException)
{
    Console.WriteLine("Invalid number!");
}

Error Handling Template

try
{
    // File operations here
}
catch (FileNotFoundException)
{
    Console.WriteLine("File not found!");
}
catch (UnauthorizedAccessException)
{
    Console.WriteLine("No permission to access file!");
}
catch (IOException ex)
{
    Console.WriteLine($"I/O error: {ex.Message}");
}
catch (Exception ex)
{
    Console.WriteLine($"Unexpected error: {ex.Message}");
}
