üîë XML Schema vs DTD

Validacija z XmlSchemaSet

static void ValidirajXML(string pot)
{
    validacijaOK = true;
    validacijskeNapake.Clear();

    try
    {
        // XmlSchemaSet za kompleksne sheme (z include/redefine)
        XmlSchemaSet schemaSet = new XmlSchemaSet();
        schemaSet.XmlResolver = new XmlUrlResolver();  // ‚ö†Ô∏è KLJUƒåNO za resolviranje include/redefine
        schemaSet.Add(null, potDoSheme);

        /* Hierarhija:
           shema.xsd
             ‚Üì include
           tipi.xsd (EmailType, DavcnaStevilkaType, ArtikelIdType, BaseArtikelType...)
             ‚Üì redefine
           omejitve.xsd (BaseDobaviteljType ‚Üí raz≈°irjen z dodatnimi elementi)
        */

        XmlReaderSettings settings = new XmlReaderSettings();
        settings.ValidationType = ValidationType.Schema;
        settings.ValidationEventHandler += ValidationCallBack;
        settings.Schemas = schemaSet;
        settings.XmlResolver = new XmlUrlResolver();  // ‚ö†Ô∏è KLJUƒåNO

        using (XmlReader reader = XmlReader.Create(pot, settings))
        {
            Console.WriteLine($"Preverjam: {pot}");
            while (reader.Read());  // Prebere cel dokument, validacija poteka sproti
        }

        if (validacijaOK)
        {
            Console.WriteLine($"‚úÖ Dokument {pot} je veljaven!");
        }
        else
        {
            Console.WriteLine($"‚ùå Dokument {pot} vsebuje napake:");
            foreach (string napaka in validacijskeNapake)
            {
                Console.WriteLine($"  - {napaka}");
            }
        }
    }
    catch (Exception ex)
    {
        Console.WriteLine($"‚ùå Kritiƒçna napaka pri validaciji: {ex.Message}");
        validacijaOK = false;
    }
}

private static void ValidationCallBack(object sender, ValidationEventArgs e)
{
    string napaka = $"{e.Severity}: {e.Message}";
    validacijskeNapake.Add(napaka);
    Console.WriteLine($"  ‚ùå {napaka}");
    validacijaOK = false;
}

<!-- RESTRICTION - Omeji tip (bolj strogo) -->
<xs:simpleType name="OmejenaZaloga">
    <xs:restriction base="xs:int">
        <xs:minInclusive value="0"/>      <!-- Minimalno 0 -->
        <xs:maxInclusive value="1000"/>   <!-- Maksimalno 1000 -->
    </xs:restriction>
</xs:simpleType>

üîë Povzetek kljuƒçnih konceptov

  1. DTD - Enostaven, omejen, primeren za osnovno validacijo
  2. XML Schema - Moƒçnej≈°a tipizacija, veƒç omejitev, bolj fleksibilen
  3. Include/Redefine - Modularnost in ponovno uporaba tipov
  4. Regex - Za kompleksne validacijske vzorce (email, ID, davƒçna...)
  5. XmlResolver - Obvezen za resolviranje include/redefine/import
  6. Temp datoteke - Vedno validiraj pred prepisovanjem originalov
  7. InvariantCulture - Za pravilno formatiranje decimalnih ≈°tevil
  8. Null operatorji - ?. in ?? za varno branje XML-ja
